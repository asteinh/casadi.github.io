#include <casadi/casadi.hpp>

auto opti = casadi::Opti();

auto u = opti.variable(5,1);

opti.minimize(dot(u,u));
opti.subject_to(u<=2);

decltype(u) x = 1;
for (int i=0;i<5;++i) {
  x = x+cos(x)+u(i);
}
opti.subject_to(x==6);

opti.solver("ipopt");
auto sol = opti.solve();

std::cout << sol.value(u) << std::endl;
